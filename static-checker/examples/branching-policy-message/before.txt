User {
    create: public,
    delete: none,
}
Message {
    create: public,
    delete: none,
    from : Id(User) {
        read: m -> (if true then [m.from, m.to] else [m.to]),
        write: none,
    },

    to : Id(User) {
        read: m -> m.from + (if false then m.from else m.to),
        write: none,
    },

    text : String {
        read: m -> m.to + m.from,
        write: none,
    },
}
