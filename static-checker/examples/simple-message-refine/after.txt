User {}

Message {
    create: public,
    delete: none,
    from : Id(User) {
        read: m -> m.from + m.to,
        write: none,
    },

    to : Id(User) {
        read: m -> m.to + m.from,
        write: none,
    },

    text : String {
        read: m -> m.from,
        write: none,
    },
}
