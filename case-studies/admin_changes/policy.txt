@principle
User {
  create: public,
  delete: u -> [u.id],

  is_admin: Bool {
    read: public,
    write: none,
  },

  site: Id(Site) {
    read: public,
    write: none,
  },
}

Site {
  create: public,
  delete: none,

  name: String {
    read: public,
    write: s -> User::Find({site: s.id, is_admin: true}).map(u -> u.id),
  },
}
